  def cross_entropy_error(y, t):

*** 监督数据是标签形式（非one-hot表示，而是像"2", "7"这样的标签）
#+BEGIN_SRC python
  def cross_entropy_error(y, t):
      if y.ndim == 1:
          y = y.reshape(1, y.size)
          t = t.reshape(1, t.size)

      size = y.shape[0]
      return -np.sum(
          np.log(y[np.arange(size), t] + 1e-7)) / batch_size
#+END_SRC
