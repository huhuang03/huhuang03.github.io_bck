** div
#+BEGIN_SRC python
  def divide(x, y):
      if y > x:
          return 0
      q = divide(x, y << 1)
      if (x - 2 * q * y) < y:
          return 2 * q
      else:
          return 2 * q + 1

#+END_SRC
